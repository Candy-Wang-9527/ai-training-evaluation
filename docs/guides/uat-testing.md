## 🐛 常见问题和调试

### 问题1: 按钮点击没反应

**可能原因:**
1. JavaScript文件未正确加载
2. DOM元素ID不匹配
3. 事件监听器未正确绑定
4. 有JavaScript错误阻止执行

**调试步骤:**
```javascript
// 1. 检查元素是否存在
console.log('按钮元素:', document.getElementById('testConnectionBtn'));

// 2. 检查函数是否存在
console.log('函数类型:', typeof testBaseConnection);

// 3. 检查事件监听器
const btn = document.getElementById('testConnectionBtn');
if (btn) {
    console.log('事件监听器:', getEventListeners(btn));
}

// 4. 手动触发函数
try {
    await testBaseConnection();
} catch (error) {
    console.error('函数执行错误:', error);
}
```

### 问题2: 滑块不同步

**调试步骤:**
```javascript
// 检查滑块和输入框元素
const slider = document.getElementById('score1Slider');
const input = document.getElementById('score1Input');

console.log('滑块存在:', !!slider);
console.log('输入框存在:', !!input);

// 检查事件监听器
if (slider && input) {
    console.log('滑块事件:', getEventListeners(slider));
    console.log('输入框事件:', getEventListeners(input));
}

// 测试滑块功能
if (slider) {
    slider.value = 80;
    slider.dispatchEvent(new Event('input'));
    console.log('输入框值:', input.value);
}
```

### 问题3: 配置保存失败

**调试步骤:**
```javascript
// 检查localStorage
console.log('Base配置:', localStorage.getItem('feishuBaseConfig'));
console.log('角色配置:', localStorage.getItem('user_role_mappings'));

// 手动测试保存

    const config = {
        appToken: 'test',
        baseUrl: 'test',
        // ...其他配置
    };
    localStorage.setItem('feishuBaseConfig', JSON.stringify(config));
    console.log('手动保存成功');
} catch (error) {
    console.error('保存失败:', error);
}
```
